# 4) Minimal HTML + Vanilla JS example to open side menu and load correct toppings

Paste this into your site where your menu list exists. It assumes each front-page item has `data-item-id` or `data-type`. You can adapt for React/Vue quickly.

```html
<!-- Side panel -->
<div id="toppingsSide" class="side-panel hidden" aria-hidden="true">
  <button id="closeSide">Close</button>
  <h2 id="sideTitle"></h2>
  <div id="hotSection"></div>
  <div id="coldSection"></div>
  <div id="saucesSection"></div>
  <label><input type="checkbox" id="addFries"> Add Fries</label>
  <button id="addToOrder">Add to Order</button>
</div>

<script>
const menuData = /* paste the JSON above as JS object */ {
  toppings: { hot: ["Grilled Chicken","Sumac Chicken","Skirt Steak","Saffron Rice","Coconut Rice","Roasted Cauliflower","Roasted Corn","Black Beans","Quinoa"], cold: ["Lettuce","Pickled Cabbage","Pico de Gallo","Cucumber Tomato Salsa","Avocado","Guacamole","Pineapple","Banana","Watermelon"], sauces: ["Chipotle Sauce","Harissa","Jalapeño Mayo","Ranch","Zawadi Sauce","Tahini","Lime Squeeze"] },
  menuItems: [ {id:"fruit-bowl", name:"Fruit Bowl", type:"fruit", allowedToppings:["cold","sauces"]}, {id:"chicken-bowl", name:"Chicken Bowl", type:"chicken", allowedToppings:["hot","cold","sauces","fries"]}, {id:"beef-bowl", name:"Beef Bowl", type:"beef", allowedToppings:["hot","cold","sauces","fries"]}, {id:"blackbean-bowl", name:"Black Bean Bowl", type:"veg", allowedToppings:["hot","cold","sauces","fries"]}, {id:"chicken-quesadilla", name:"Chicken Quesadilla", type:"quesadilla", baseProtein:"chicken", allowedToppings:["hot","cold","sauces","fries"] }]
};

function openSideMenu(itemId){
  const item = menuData.menuItems.find(m=>m.id===itemId);
  if(!item) return;
  document.getElementById('sideTitle').innerText = item.name;
  // Clear sections
  ['hotSection','coldSection','saucesSection'].forEach(id=>document.getElementById(id).innerHTML='');
  // Render allowed toppings
  if(item.allowedToppings.includes('hot')){
    const hot = menuData.toppings.hot.map(t => `<label><input type="checkbox" name="hot" value="${t}">${t}</label>`).join('');
    document.getElementById('hotSection').innerHTML = `<h3>Hot</h3>${hot}`;
  }
  if(item.allowedToppings.includes('cold')){
    const cold = menuData.toppings.cold.map(t => `<label><input type="checkbox" name="cold" value="${t}">${t}</label>`).join('');
    document.getElementById('coldSection').innerHTML = `<h3>Cold</h3>${cold}`;
  }
  if(item.allowedToppings.includes('sauces')){
    const s = menuData.toppings.sauces.map(t => `<label><input type="checkbox" name="sauce" value="${t}">${t}</label>`).join('');
    document.getElementById('saucesSection').innerHTML = `<h3>Sauces</h3>${s}`;
  }

  // Quesadilla rules: if quesadilla, auto-check double meat and lock it
  if(item.type==='quesadilla'){
    // find checkboxes that match baseProtein
    setTimeout(()=>{ // allow DOM to render
      document.querySelectorAll('input[name="hot"]').forEach(cb=>{
        if(cb.value.toLowerCase().includes(item.baseProtein)){
          cb.checked = true;
          cb.disabled = true; // lock double-meat selection
        }
      });
      // guacamole on the side required: mark it visually (you can require it at order time)
    },10);
  }

  // show fries checkbox according to allowed
  document.getElementById('addFries').style.display = item.allowedToppings.includes('fries') ? 'inline-block' : 'none';

  // show panel
  document.getElementById('toppingsSide').classList.remove('hidden');
  document.getElementById('toppingsSide').setAttribute('aria-hidden','false');
}

// Hook front page items: ensure every clickable has data-item-id and an event listener
document.querySelectorAll('.menu-item').forEach(el=>{
  // ensure it's clickable: replace non-interactive element with a button if needed.
  el.addEventListener('click', e=>{
    const id = el.dataset.itemId;
    openSideMenu(id);
  });
});

document.getElementById('closeSide').addEventListener('click', ()=>{
  document.getElementById('toppingsSide').classList.add('hidden');
  document.getElementById('toppingsSide').setAttribute('aria-hidden','true');
});
</script>

<style>
.side-panel{ position:fixed; right:0; top:0; width:340px; height:100%; background:#fff; box-shadow:-8px 0 20px rgba(0,0,0,.15); padding:16px; z-index:1000;}
.side-panel.hidden{ display:none; }
.menu-item{ cursor:pointer; } /* ensure pointer shows */
</style>
```

**Notes:**

* Replace `.menu-item` and `data-item-id` usage to match your site’s actual DOM selectors.
* If you use React/Vue, implement the same mapping logic in state and open a slide-over component.

# 5) How to fix “not all items on the front page are clickable”

* Inspect the front page elements: if they’re `<div>`s, wrap content in `<button class="menu-item" data-item-id="...">` or add `tabindex="0"` and keyboard handlers.
* Make sure event listeners are attached **after** the DOM loads (or use event delegation on a parent).
* For accessibility, include `aria-label` and role attributes.
